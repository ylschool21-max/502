import React, { useEffect, useRef, useState } from 'react'; import { Hands, Results } from '@mediapipe/hands'; import { Camera } from '@mediapipe/camera_utils'; import { motion } from 'motion/react'; import { X, Trophy, Timer } from 'lucide-react'; import confetti from 'canvas-confetti'; interface GameProps { onEnd: (score: number) => void; onBack: () => void; } type GameObject = { id: number; x: number; y: number; type: 'fruit' | 'obstacle'; emoji: string; speed: number; size: number; }; const FRUITS = [ { emoji: '🍅', points: 1 }, ]; const OBSTACLES = [ { emoji: '🪨', points: -2 }, { emoji: '🍃', points: -1 }, ]; export default function Game({ onEnd, onBack }: GameProps) { const videoRef = useRef(null); const canvasRef = useRef(null); const [score, setScore] = useState(0); const [timeLeft, setTimeLeft] = useState(60); const [isReady, setIsReady] = useState(false); const gameObjects = useRef([]); const handPos = useRef<{ x: number; y: number } | null>(null); const requestRef = useRef(null); const scoreRef = useRef(0); // Sound effects (using Audio API) const playPopSound = () => { const ctx = new (window.AudioContext || (window as any).webkitAudioContext)(); const osc = ctx.createOscillator(); const gain = ctx.createGain(); osc.type = 'sine'; osc.frequency.setValueAtTime(440, ctx.currentTime); osc.frequency.exponentialRampToValueAtTime(880, ctx.currentTime + 0.1); gain.gain.setValueAtTime(0.1, ctx.currentTime); gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.1); osc.connect(gain); gain.connect(ctx.destination); osc.start(); osc.stop(ctx.currentTime + 0.1); }; const playBadSound = () => { const ctx = new (window.AudioContext || (window as any).webkitAudioContext)(); const osc = ctx.createOscillator(); const gain = ctx.createGain(); osc.type = 'sawtooth'; osc.frequency.setValueAtTime(220, ctx.currentTime); osc.frequency.linearRampToValueAtTime(110, ctx.currentTime + 0.2); gain.gain.setValueAtTime(0.1, ctx.currentTime); gain.gain.linearRampToValueAtTime(0.01, ctx.currentTime + 0.2); osc.connect(gain); gain.connect(ctx.destination); osc.start(); osc.stop(ctx.currentTime + 0.2); }; useEffect(() => { if (!videoRef.current || !canvasRef.current) return; const hands = new Hands({ locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`, }); hands.setOptions({ maxNumHands: 1, modelComplexity: 1, minDetectionConfidence: 0.5, minTrackingConfidence: 0.5, }); hands.onResults((results: Results) => { if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) { // Use the palm center (landmark 0 or average of 0, 5, 9, 13, 17) const landmarks = results.multiHandLandmarks[0]; const x = (1 - landmarks[9].x) * canvasRef.current!.width; const y = landmarks[9].y * canvasRef.current!.height; handPos.current = { x, y }; } else { handPos.current = null; } }); const camera = new Camera(videoRef.current, { onFrame: async () => { await hands.send({ image: videoRef.current! }); }, width: 640, height: 480, }); camera.start().then(() => setIsReady(true)); return () => { camera.stop(); hands.close(); }; }, []); useEffect(() => { const timer = setInterval(() => { setTimeLeft((prev) => { if (prev <= 1) { clearInterval(timer); if (scoreRef.current >= 20) { onEnd(scoreRef.current); } else { alert('시간 종료! 20개를 모으지 못했어요.'); onBack(); } return 0; } return prev - 1; }); }, 1000); return () => clearInterval(timer); }, [onEnd, onBack]); const playSuccessSound = () => { const ctx = new (window.AudioContext || (window as any).webkitAudioContext)(); // Cheering sound (simulated with noise and multiple oscillators) for (let i = 0; i < 5; i++) { const osc = ctx.createOscillator(); const gain = ctx.createGain(); osc.type = 'sine'; osc.frequency.setValueAtTime(400 + Math.random() * 400, ctx.currentTime); gain.gain.setValueAtTime(0, ctx.currentTime); gain.gain.linearRampToValueAtTime(0.1, ctx.currentTime + 0.1); gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 1); osc.connect(gain); gain.connect(ctx.destination); osc.start(); osc.stop(ctx.currentTime + 1); } }; const update = () => { if (!canvasRef.current) return; const canvas = canvasRef.current; const ctx = canvas.getContext('2d'); if (!ctx) return; ctx.clearRect(0, 0, canvas.width, canvas.height); // Spawn objects if (Math.random() < 0.05) { const isFruit = Math.random() > 0.15; // Reduced obstacles by half (from 0.3 to 0.15) const list = isFruit ? FRUITS : OBSTACLES; const item = list[Math.floor(Math.random() * list.length)]; gameObjects.current.push({ id: Date.now() + Math.random(), x: Math.random() * canvas.width, y: -50, type: isFruit ? 'fruit' : 'obstacle', emoji: item.emoji, speed: 2 + Math.random() * 10, // More diverse speed for agility (2 to 12) size: isFruit ? (60 + Math.random() * 30) : (40 + Math.random() * 20), // Larger cherry tomatoes }); } // Update and draw objects gameObjects.current = gameObjects.current.filter((obj) => { obj.y += obj.speed; // Collision detection if (handPos.current) { const dx = obj.x - handPos.current.x; const dy = obj.y - handPos.current.y; const dist = Math.sqrt(dx * dx + dy * dy); if (dist < obj.size) { if (obj.type === 'fruit') { scoreRef.current += 1; setScore(scoreRef.current); playPopSound(); // Check success condition immediately if (scoreRef.current === 20) { playSuccessSound(); confetti({ particleCount: 150, spread: 100, origin: { y: 0.6 } }); setTimeout(() => onEnd(scoreRef.current), 1500); } } else { scoreRef.current = Math.max(0, scoreRef.current - 2); setScore(scoreRef.current); playBadSound(); } return false; // Remove object } } // Draw ctx.font = `${obj.size}px Arial`; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(obj.emoji, obj.x, obj.y); return obj.y < canvas.height + 100; }); // Draw hand indicator if (handPos.current) { ctx.beginPath(); ctx.arc(handPos.current.x, handPos.current.y, 20, 0, Math.PI * 2); ctx.fillStyle = 'rgba(255, 77, 77, 0.5)'; ctx.fill(); ctx.strokeStyle = 'white'; ctx.lineWidth = 3; ctx.stroke(); } requestRef.current = requestAnimationFrame(update); }; useEffect(() => { requestRef.current = requestAnimationFrame(update); return () => { if (requestRef.current) cancelAnimationFrame(requestRef.current); }; }, []); return (
{/* Camera Feed (Hidden or mirrored background) */}
); }
